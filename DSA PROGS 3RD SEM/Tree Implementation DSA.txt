#include<stdio.h>
#include<stdlib.h>
#include<time.h>
#include<unistd.h>
#include<string.h>
#include<conio.h>
typedef struct node
{
	int data;
	struct node *left;
	struct node *right;
} node;
node *create()
{
	node *p;
	int x;
	printf("Enter the value for the node or -1 for no new node:\n");
	scanf("%d",&x);
	if(x==-1)
	{
		return NULL;
	}
	p=(struct node*)malloc(sizeof(struct node));
	p->data=x;
	printf("\nEnter the left child of %d:\n",x);
	p->left=create();
	printf("\nEnter the right child of %d:\n",x);
	p->right=create();
	return p;
}
void preorder(node *t)
{
	if(t!=NULL)
	{
		printf("%d\n",t->data);
		preorder(t->left);
		preorder(t->right);
	}
}
void inorder(node *t)
{
	if(t!=NULL)
	{
		inorder(t->left);
		printf("%d\n",t->data);
		inorder(t->right);
	}
}
void postorder(node *t)
{
	if(t!=NULL)
	{
		inorder(t->left);
		inorder(t->right);
		printf("%d\n",t->data);
	}
}
int main()
{
	time_t now =time(NULL);
	char *str=ctime(&now);
	printf("\nGMT +5.30hrs\n");
	printf("%s\n",str);
	int n;
	node *root;
	root=create();
	while(1)
	{
		printf("\n1.Preorder\n");
		printf("\n2.Inorder\n");
		printf("\n3.Postorder\n");
		printf("\n4.Exit\n");
		printf("\nEnter your choice:\n");
		scanf("%d",&n);
		switch(n)
		{
			case 1:
				printf("\nThe preorder traversal is\n");
				preorder(root);
				break;
			case 2:
				printf("\nThe inorder traversal is\n");
				inorder(root);
				break;
			case 3:
				printf("\nThe postorder traversal is\n");
				postorder(root);
				break;
			case 4:
				printf("\nExiting the code!\n");
				exit(0);
			default:
				printf("Wrong option entered!\n");
				break;
		}
	}
	return 0;
}
